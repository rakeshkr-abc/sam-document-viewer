const COMM_FOLDER_ID = 'YOUR_COMMUNICATION_FOLDER_ID_HERE';
const TECH_FOLDER_ID = 'YOUR_TECHNICAL_FOLDER_ID_HERE';

function exportToJSON() {
  const commFiles = [];
  const commFolder = DriveApp.getFolderById(COMM_FOLDER_ID);
  const commIterator = commFolder.getFiles();
  
  while (commIterator.hasNext()) {
    const file = commIterator.next();
    commFiles.push({
      name: file.getName(),
      id: file.getId()
    });
  }
  
  const techFiles = [];
  const techFolder = DriveApp.getFolderById(TECH_FOLDER_ID);
  const techIterator = techFolder.getFiles();
  
  while (techIterator.hasNext()) {
    const file = techIterator.next();
    techFiles.push({
      name: file.getName(),
      id: file.getId()
    });
  }
  
  const jsonData = {
    communication: commFiles,
    technical: techFiles
  };
  
  const jsonBlob = Utilities.newBlob(
    JSON.stringify(jsonData, null, 2),
    'application/json',
    'api.json'
  );
  
  DriveApp.createFile(jsonBlob);
  console.log('✅ api.json CREATED! Check Google Drive root');
  console.log(`📊 Communication: ${commFiles.length} files`);
  console.log(`📊 Technical: ${techFiles.length} files`);
}

function bulkFixPermissions() {
  const commFolder = DriveApp.getFolderById(COMM_FOLDER_ID);
  const techFolder = DriveApp.getFolderById(TECH_FOLDER_ID);
  
  let fixed = 0;
  
  const commFiles = commFolder.getFiles();
  while (commFiles.hasNext()) {
    commFiles.next().setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    fixed++;
  }
  
  const techFiles = techFolder.getFiles();
  while (techFiles.hasNext()) {
    techFiles.next().setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    fixed++;
  }
  
  console.log(`✅ FIXED ${fixed} files - All PUBLIC!`);
}
